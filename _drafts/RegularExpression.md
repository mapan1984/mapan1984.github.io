### 元字符

    . ^ $ * + ? { [ ] } \ | ( )

    ^ 行开头
    $ 行结束

### 字符匹配

    .  (出换行外的)任意字符
    \d [0-9]
    \D [^0-9]
    \s [\t\n\r\f\v] 任何空白字符
    \S [^\t\n\r\f\v]
    \w [a-zA-Z0-9]
    \W [^a-zA-Z0-9]

### 重复
    
    {m,n}

    * {0,}
    + {1,}
    ? {0,1}

A or B  A|B

### 贪婪匹配

    var re = /^(\d+)(0*)$/;
    re.exec('102300'); //('102300', '102300', '')

### 非贪婪匹配

    var re = /^(\d+?)(0*)$/;
    re.exec('102300'); //('102300', '1023', '00')
