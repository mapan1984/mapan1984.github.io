``` python
# 属性attr: 0表示无属性，1表示高亮，4加下划线，5表示闪烁 ，7反显 ，8消隐 
# 字颜色fore:30———–39 30:黑 31:红 32:绿 33:黄 34:蓝色 35:紫色 36:深绿 37:白色
# 字背景颜色back:40—-49 40:黑 41:深红 42:绿 43:黄色 44:蓝色 45:紫色 46:深绿 47:白色
for atrr in [0,1,4,5,7]:   
    print("attribute %d ------------------------------" % atrr)
    for fore in [30,31,32,33,34,35,36,37]:
        for back in [40,41,42,43,44,45,46,47]:
            #除了atrr，分别代表 字的颜色，底色
            color = "\x1B[%d;%d;%dm" % (atrr,fore,back)  
            print("%s %d-%d-%d\x1B[0m" % (color,atrr,fore,back))
        print()
```
